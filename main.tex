\documentclass[10pt]{article}

%% Language and font encodings
\usepackage[english]{babel}
%\usepackage[utf8x]{inputenc}
\usepackage[T1]{fontenc}

%% set font
\renewcommand{\familydefault}{\sfdefault}
\usepackage{lmodern}


%% Sets page size and margins
\usepackage[a4paper,landscape,top=.2cm,bottom=.2cm,left=.2cm,right=.2cm,marginparwidth=.75cm]{geometry}


%% change following parameters to make your poster fit a single page
\usepackage{titlesec}
\titlespacing{\section}{0pt}{5pt}{3pt} %change spacing between section title and text. {Left space}{Above}{Below}
\linespread{0.01} %adjust line spacing
\usepackage{scrextend}
\changefontsizes[10pt]{5pt} %I use this all the time to all fit
\setlength{\parindent}{0pt}%remove paragraph indent
\setlength{\parskip}{1.5pt}%adjust line spacing between paragraphs

%% Set up multicolumnness

\usepackage{multicol}
\usepackage{lipsum}% dummy text
\setlength{\columnseprule}{0.4pt} %comment this out if you don't like the lines separating columns
\newenvironment{Figure} %this is to insert figures in columns
  {\par\medskip\noindent\minipage{\linewidth}}
  {\endminipage\par\medskip}

%% Useful packages
\usepackage{amsmath}
\usepackage{graphicx}
\usepackage[colorinlistoftodos]{todonotes}
\usepackage[colorlinks=true, allcolors=black]{hyperref}
\usepackage{caption}
\usepackage{subcaption}
\usepackage{hyperref}
\usepackage{xcolor}%Use this to color text as needed
\usepackage{tikzpagenodes} % use this to add logos or other decorations

%% List of sexi colors

\definecolor{blue}{rgb}{0.3, 0.4, 0.8}
\definecolor{green}{rgb}{0.3, 0.7, 0.4}
\definecolor{red}{rgb}{0.7, 0.4, 0.3}

%% Bibliography

\usepackage[style=science]{biblatex} % Hey, I'm no luxury journal fanboy. Only using this because they have a concise bibliography format, which works great for posters.
\bibliography{References}
\usepackage{csquotes}
